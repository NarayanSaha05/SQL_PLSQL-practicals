Q4. Write a PL/SQL program using FOR loop to insert ten rows into a database table.

Q7. Write a PL/SQL Program to display a given number in Reverse Order.
DECLARE
    n int;
    rev int := 0;
BEGIN
    n := &n;

    WHILE n > 0 LOOP
        rev := rev * 10 + MOD(n, 10);
        n := n / 10;
    END LOOP;

    dbms_output.put_line('Reversed Number = ' || rev);
END;
/

Q14. Create a procedure which will accept Deptno and Display no of employees under different grade.
->
CREATE OR REPLACE PROCEDURE emp_details(
	p_deptno IN NUMBER
) IS
BEGIN
	FOR r IN (
		SELECT s.grade, count(*) cnt
		from emp e, salgrade s
		where e.deptno = p_deptno
		GROUP BY s.grade
		ORDER BY s.grade
	) LOOP 
		DBMS_OUTPUT.PUT_LINE(r.grade || r.cnt);
	END LOOP;
END;
/

Q15. Write a function total_salary_dept that returns the total salary of all employees in a given department.
->
CREATE OR REPLACE FUNCTION total_salary_dept(
	dno IN NUMBER
)
RETURN NUMBER
IS
	total_sal NUMBER;
BEGIN
	SELECT sum(sal)
	INTO total_sal
	FROM emp
	where deptno = dno;
	
	return total_sal;
END;
/

Q17. Write a function calc_bonus that returns the bonus for an employee as 20% of salary.
->
CREATE OR REPLACE FUNCTION cal_bonus(
	p_no IN NUMBER
)
RETURN NUMBER
IS
	new_sal emp.sal%type;
BEGIN

	SELECT sal
	into new_sal
	from emp
	where empno = p_no;

	return new_sal * 0.20;
END;

Q18. Write a procedure show_emp that accepts an employee ID and prints: Name, Salary & DeptName.
->
CREATE OR REPLACE PROCEDURE show_emp(
	e_id IN NUMBER
) IS
	v_name emp.ename%type;
	v_sal emp.sal%type;
	v_dept emp.dname%type;
BEGIN
	SELECT e.ename, e.sal, d.dname
	INTO v_name, v_sal, v_dept
	from emp e, dept d
	where e.deptno = d.deptno
	AND e.empno = e_id
	
	DBMS_OUTPUT.PUT_LINE(v_name || v_sal || v_dept);
END;
/

Q19. Create a procedure update_salary that increases the salary of a given employee by a given percentage. Parameters: employee ID, percentage.
->
CREATE OR REPLACE PROCEDURE update_salary(
	eno IN NUMBER,
	percentage IN NUMBER
)
BEGIN
	UPDATE emp
	SET sal = sal + (sal * percent / 100)
	where empno = eno;
END;
/


Q20. Write a function calculate_age(p_emp_id NUMBER) that retrieves the employee's DOB, calculates age in years, and returns the age.
->
CREATE OR REPLACE FUNCTION calculate_age(
	p_emp_id IN NUMBER
) IS
	v_dob DATE;
	v_age NUMBER;
BEGIN
	SELECT dob
	INTO v_dob
	from emp
	where empno = p_emp_id;

	v_age := FLOOR(MONTHS_BETWEEN(SYSDATE, v_dob) / 12);

	return v_age;
END;
/

Q23. Using a cursor FOR loop, increase the salary of employees in the SALES department by 10%. Also display the old salary and new salary.
->

  CURSOR c IS SELECT * FROM employee WHERE dept_id = 30;
BEGIN
  FOR r IN c LOOP
    DBMS_OUTPUT.PUT_LINE('Old Salary: ' || r.salary);

    UPDATE employee 
    SET salary = salary * 1.1 
    WHERE emp_id = r.emp_id;

  END LOOP;
END;

CREATE OR REPLACE TRIGGER trg_salary_audit
BEFORE UPDATE OF salary ON employee
FOR EACH ROW
BEGIN
  INSERT INTO salary_log(emp_id, old_salary, new_salary, change_date)
  VALUES(:OLD.emp_id, :OLD.salary, :NEW.salary, SYSDATE);
END;
/
Input-------
UPDATE employee
SET salary = 60000
WHERE emp_id = 1;

output-----
1 row updated.
check------

CREATE OR REPLACE TRIGGER trg_no_salary_decrease
BEFORE UPDATE OF salary ON employee
FOR EACH ROW
BEGIN
  IF :NEW.salary < :OLD.salary THEN
    RAISE_APPLICATION_ERROR(-20001, 'Salary cannot be decreased');
  END IF;
END;
/
Input----
UPDATE employee
SET salary = 30000
WHERE emp_id = 1;
Output-----
ERROR at line 1:
ORA-20001: Salary cannot be decreased

BEFORE INSERT ON student
FOR EACH ROW

BEGIN
  SELECT COUNT(*) INTO c FROM student WHERE email=:NEW.email;

    RAISE_APPLICATION_ERROR(-20001,'duplicate email');
  END IF;
END;
