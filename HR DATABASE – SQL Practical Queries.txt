HR DATABASE – SQL Practical Queries
Database Schema
-- Drop and Create EMP table
DROP TABLE IF EXISTS emp;
CREATE TABLE emp (
empno decimal(4,0) NOT NULL,
ename varchar(10) default NULL,
job varchar(9) default NULL,
mgr decimal(4,0) default NULL,
hiredate date default NULL,
sal decimal(7,2) default NULL,
comm decimal(7,2) default NULL,
deptno decimal(2,0) default NULL
);
-- Drop and Create DEPT table
DROP TABLE IF EXISTS dept;
CREATE TABLE dept (
deptno decimal(2,0) default NULL,
dname varchar(14) default NULL,
loc varchar(13) default NULL
);
-- Create SALGRADE table
CREATE TABLE salgrade (
GRADE INT NOT NULL PRIMARY KEY,
LOSAL INT NOT NULL,
HISAL INT NOT NULL
);
I. Basic SQL Queries (Easy Level)
Q1. Display all details of employees.
SELECT * FROM emp;
Q2. Display employee name and salary.
SELECT ename, sal FROM emp;
Q3. Display employee name and department number.
SELECT ename, deptno FROM emp;
Q4. Display distinct job titles from the EMP table.
SELECT DISTINCT job FROM emp;
Q5. Display employees who belong to department number 30.
SELECT * FROM emp WHERE deptno = 30;
Q6. Display employees whose salary is greater than 2000.
SELECT * FROM emp WHERE sal > 2000;
Q7. Display employees hired after 01-JAN-1981.
SELECT * FROM emp WHERE hiredate > '1981-01-01';
Q8. Display employees whose name starts with the letter S.
SELECT * FROM emp WHERE ename LIKE 'S%';
Q9. Display employees who receive any commission.
SELECT * FROM emp WHERE comm IS NOT NULL AND comm > 0;
Q10. Display employees whose department number is 10 or 20.
SELECT * FROM emp WHERE deptno IN (10, 20);
Q11. Display all employees sorted by salary in descending order.
SELECT * FROM emp ORDER BY sal DESC;
Q12. Display employee name, salary and commission sorted by name.
SELECT ename, sal, comm FROM emp ORDER BY ename;
Q13. Display employees who have no manager.
SELECT * FROM emp WHERE mgr IS NULL;
Q14. Display the location of department number 20.
SELECT loc FROM dept WHERE deptno = 20;
Q15. Display department details sorted by department name in alphabetical order.
SELECT * FROM dept ORDER BY dname ASC;
Q16. Display employees whose salary is between 1000 and 3000.
SELECT * FROM emp WHERE sal BETWEEN 1000 AND 3000;
Q17. Count the total number of employees in the EMP table.
SELECT COUNT(*) AS total_employees FROM emp;
Q18. Display the minimum and maximum salary from the EMP table.
SELECT MIN(sal) AS min_salary, MAX(sal) AS max_salary FROM emp;
Q19. Display total number of employees for each job title.
SELECT job, COUNT(*) AS total FROM emp GROUP BY job;
Q20. Display the current system date in MySQL.
SELECT CURDATE();
II. Conditional, Sorting, Date and Aggregate Queries (Moderate Level)
Q21. Display employee name, salary and yearly salary (salary * 12).
SELECT ename, sal, sal * 12 AS yearly_salary FROM emp;
Q22. Display employees who joined in the year 1981.
SELECT * FROM emp WHERE YEAR(hiredate) = 1981;
Q23. Display employee details sorted by hire date in descending order.
SELECT * FROM emp ORDER BY hiredate DESC;
Q24. Display the average salary of each department.
SELECT deptno, AVG(sal) AS avg_salary FROM emp GROUP BY deptno;
Q25. Display employee name along with department name using a join operation.
SELECT e.ename, d.dname FROM emp e JOIN dept d ON e.deptno = d.deptno;
Q26. Display employees earning more than the average salary of the company.
SELECT * FROM emp WHERE sal > (SELECT AVG(sal) FROM emp);
Q27. Display employees who have commission in each department.
SELECT deptno, ename, comm FROM emp WHERE comm IS NOT NULL AND comm > 0 ORDER BY deptno;
Q28. Display employees whose name contains the character A (case-insensitive).
SELECT * FROM emp WHERE ename LIKE '%A%';
Q29. Display the highest salary of each job title.
SELECT job, MAX(sal) AS max_salary FROM emp GROUP BY job;
Q30. Display employees who work in the department located in NEW YORK.
SELECT e.* FROM emp e JOIN dept d ON e.deptno = d.deptno WHERE d.loc = 'NEW YORK';
Q31. Display employees who were hired on a Monday.
SELECT * FROM emp WHERE DAYNAME(hiredate) = 'Monday';
Q32. Count the number of employees who do not receive commission.
SELECT COUNT(*) AS no_commission FROM emp WHERE comm IS NULL OR comm = 0;
Q33. Display job titles along with the total commission paid for each job.
SELECT job, SUM(comm) AS total_commission FROM emp GROUP BY job;
Q34. Display departments having more than 3 employees.
SELECT deptno, COUNT(*) AS emp_count FROM emp GROUP BY deptno HAVING COUNT(*) > 3;
Q35. Display the employees who draw the second highest salary.
SELECT * FROM emp WHERE sal = (SELECT MAX(sal) FROM emp WHERE sal < (SELECT MAX(sal) FROM emp));
Q36. Display employees who have the same job title as SMITH.
SELECT * FROM emp WHERE job = (SELECT job FROM emp WHERE ename = 'SMITH') AND ename != 'SMITH';
Q37. Display each employee's salary grade using the SALGRADE table.
SELECT e.ename, e.sal, s.grade FROM emp e JOIN salgrade s ON e.sal BETWEEN s.losal AND s.hisal;
Q38. Display employee name, salary and salary grade together.
SELECT e.ename, e.sal, s.grade FROM emp e JOIN salgrade s ON e.sal BETWEEN s.losal AND s.hisal ORDER BY s.grade;
Q39. Display employees who are managers.
SELECT DISTINCT e.* FROM emp e WHERE e.empno IN (SELECT mgr FROM emp WHERE mgr IS NOT NULL);
Q40. Display the employee who earns the highest salary in department 30.
SELECT * FROM emp WHERE deptno = 30 AND sal = (SELECT MAX(sal) FROM emp WHERE deptno = 30);
III. Subqueries, Joins, Views, DDL, DML and Advanced SQL Concepts (Advanced Level)
Q41. Display the employee(s) who earn the highest salary in the company.
SELECT * FROM emp WHERE sal = (SELECT MAX(sal) FROM emp);
Q42. Display employees who earn more than the average salary of their department.
SELECT e.* FROM emp e WHERE e.sal > (SELECT AVG(sal) FROM emp WHERE deptno = e.deptno);
Q43. Display the department having the maximum total salary paid to employees.
SELECT deptno, SUM(sal) AS total_sal FROM emp GROUP BY deptno ORDER BY total_sal DESC LIMIT 1;
Q44. Display employees who joined before their respective manager joined.
SELECT e.* FROM emp e JOIN emp m ON e.mgr = m.empno WHERE e.hiredate < m.hiredate;
Q45. Display department details where no employees are working.
SELECT * FROM dept WHERE deptno NOT IN (SELECT DISTINCT deptno FROM emp WHERE deptno IS NOT NULL);
Q46. Display the third highest salary using a subquery.
SELECT MIN(sal) FROM emp WHERE sal IN (SELECT DISTINCT sal FROM emp ORDER BY sal DESC LIMIT 3);
Q47. Display the count of employees hired month-wise.
SELECT MONTH(hiredate) AS month, MONTHNAME(hiredate) AS month_name, COUNT(*) AS emp_count FROM emp GROUP BY MONTH(hiredate) ORDER BY month;
Q48. Display employees who joined after the employee FORD joined.
SELECT * FROM emp WHERE hiredate > (SELECT hiredate FROM emp WHERE ename = 'FORD');
Q49. Display the employees whose department number is not present in the DEPT table.
SELECT * FROM emp WHERE deptno NOT IN (SELECT deptno FROM dept);
Q50. Display job titles along with the difference between maximum and minimum salary of that job.
SELECT job, MAX(sal) - MIN(sal) AS sal_difference FROM emp GROUP BY job;
Q51. Display employee names along with the length of each name.
SELECT ename, LENGTH(ename) AS name_length FROM emp;
Q52. Display salaries formatted to two decimal places.
SELECT ename, FORMAT(sal, 2) AS formatted_salary FROM emp;
Q53. Display each employee name masked after the first three characters (example: SMI****).
SELECT ename, CONCAT(SUBSTRING(ename, 1, 3), REPEAT('*', LENGTH(ename) - 3)) AS masked_name FROM emp;
Q54. Display employee name and commission, replacing NULL values with the text 'No Commission'.
SELECT ename, IFNULL(CAST(comm AS CHAR), 'No Commission') AS commission FROM emp;
Q55. Create a view to display employee name, salary and department name.
CREATE VIEW emp_dept_view AS SELECT e.ename, e.sal, d.dname FROM emp e JOIN dept d ON e.deptno = d.deptno;
Q56. Update the salary of employees in department 30 by increasing it by 10 percent.
UPDATE emp SET sal = sal * 1.10 WHERE deptno = 30;
Q57. Delete employees who do not belong to any department.
DELETE FROM emp WHERE deptno IS NULL;
Q58. Insert a new department with department number 50 and department name 'HR'.
INSERT INTO dept (deptno, dname) VALUES (50, 'HR');
Q59. Display employees working in the same department as BLAKE.
SELECT * FROM emp WHERE deptno = (SELECT deptno FROM emp WHERE ename = 'BLAKE') AND ename != 'BLAKE';
Q60. Display employees whose job title appears more than once in the EMP table.
SELECT * FROM emp WHERE job IN (SELECT job FROM emp GROUP BY job HAVING COUNT(*) > 1);
Q61. Display employee details whose salary is higher than the maximum salary of employees in department 10.
SELECT * FROM emp WHERE sal > (SELECT MAX(sal) FROM emp WHERE deptno = 10);
Q62. Display employees who earn more than every employee in department 30.
SELECT * FROM emp WHERE sal > ALL (SELECT sal FROM emp WHERE deptno = 30);
Q63. Display departments where all employees earn a commission.
SELECT deptno FROM emp GROUP BY deptno HAVING SUM(CASE WHEN comm IS NULL OR comm = 0 THEN 1 ELSE 0 END) = 0;
Q64. Display employees whose salary matches the average salary of their own department.
SELECT e.* FROM emp e WHERE e.sal = (SELECT AVG(sal) FROM emp WHERE deptno = e.deptno);
Q65. Display job title and employees whose salary is below the average salary of that job.
SELECT e.job, e.ename, e.sal FROM emp e WHERE e.sal < (SELECT AVG(sal) FROM emp WHERE job = e.job);
Q66. Display employees who have the minimum salary in their respective job.
SELECT e.* FROM emp e WHERE e.sal = (SELECT MIN(sal) FROM emp WHERE job = e.job);
Q67. Display employees who work in a department located in the same city as the department of KING.
SELECT e.* FROM emp e JOIN dept d ON e.deptno = d.deptno WHERE d.loc = (SELECT d2.loc FROM emp e2 JOIN dept d2 ON e2.deptno = d2.deptno WHERE e2.ename = 'KING') AND e.ename != 'KING';
Q68. Display employees whose salary is not within any salary grade range.
SELECT * FROM emp WHERE sal NOT IN (SELECT e.sal FROM emp e JOIN salgrade s ON e.sal BETWEEN s.losal AND s.hisal);
Q69. Display employees who earn the second highest salary in each department.
SELECT e.* FROM emp e WHERE e.sal = (SELECT MAX(sal) FROM emp WHERE deptno = e.deptno AND sal < (SELECT MAX(sal) FROM emp WHERE deptno = e.deptno));
Q70. Display employees whose department has at least one manager working in it.
SELECT * FROM emp WHERE deptno IN (SELECT deptno FROM emp WHERE job = 'MANAGER');
Q71. Display departments where no manager is assigned based on EMP table details.
SELECT * FROM dept WHERE deptno NOT IN (SELECT DISTINCT deptno FROM emp WHERE job = 'MANAGER');
Q72. Display employees who earn a salary that falls into the highest grade in SALGRADE.
SELECT e.* FROM emp e JOIN salgrade s ON e.sal BETWEEN s.losal AND s.hisal WHERE s.grade = (SELECT MAX(grade) FROM salgrade);
Q73. Display employees who have the same salary as someone in a different department.
SELECT DISTINCT e1.* FROM emp e1 JOIN emp e2 ON e1.sal = e2.sal AND e1.deptno != e2.deptno;
Q74. Display the employee who has the earliest hire date in each department using a correlated subquery.
SELECT * FROM emp e WHERE hiredate = (SELECT MIN(hiredate) FROM emp WHERE deptno = e.deptno);
Q75. Display all departments that have less than the average number of employees compared to all departments.
SELECT deptno, COUNT(*) AS emp_count FROM emp GROUP BY deptno HAVING COUNT(*) < (SELECT AVG(cnt) FROM (SELECT COUNT(*) AS cnt FROM emp GROUP BY deptno) AS dept_counts);
Q76. Display employees whose managers work in a different location than the employee.
SELECT e.* FROM emp e JOIN emp m ON e.mgr = m.empno JOIN dept d1 ON e.deptno = d1.deptno JOIN dept d2 ON m.deptno = d2.deptno WHERE d1.loc != d2.loc;
Q77. Display employees who work in a department where the total salary is greater than all other departments.
SELECT * FROM emp WHERE deptno = (SELECT deptno FROM emp GROUP BY deptno ORDER BY SUM(sal) DESC LIMIT 1);
Q78. Display the name of the department and employees who earn a salary greater than the average salary of all managers.
SELECT d.dname, e.ename, e.sal FROM emp e JOIN dept d ON e.deptno = d.deptno WHERE e.sal > (SELECT AVG(sal) FROM emp WHERE job = 'MANAGER');
Q79. Display the department name and employee name for employees who earn more than their manager.
SELECT d.dname, e.ename, e.sal FROM emp e JOIN emp m ON e.mgr = m.empno JOIN dept d ON e.deptno = d.deptno WHERE e.sal > m.sal;
Q80. Display employee name, salary, and department name for employees whose salary is within salary grade 3.
SELECT e.ename, e.sal, d.dname FROM emp e JOIN dept d ON e.deptno = d.deptno JOIN salgrade s ON e.sal BETWEEN s.losal AND s.hisal WHERE s.grade = 3;
